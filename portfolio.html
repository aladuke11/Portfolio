// Content cycling functionality
const changeContentBtn = document.querySelector('#change-content');
const sections = document.querySelectorAll('article section');
let currentSection = 0;

changeContentBtn.addEventListener('click', function() {
    sections[currentSection].style.opacity = '0';
    setTimeout(() => {
        sections[currentSection].style.display = 'none';
        currentSection = (currentSection + 1) % sections.length;
        sections[currentSection].style.display = 'block';
        setTimeout(() => {
            sections[currentSection].style.opacity = '1';
        }, 50);
    }, 300);
});

‚ö° Accessibility Toggle System

Site-wide animation toggle with localStorage persistence and system preference detection

View JavaScript Code 
// Neon Effects Toggle with Persistence
const neonToggle = document.querySelector('#neon-toggle');
const body = document.body;

// Check for saved preference or system preference
const savedPreference = localStorage.getItem('neonEffectsEnabled');
const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;

let neonEnabled = true;
if (savedPreference !== null) {
    neonEnabled = savedPreference === 'true';
} else if (prefersReducedMotion) {
    neonEnabled = false;
}

// Apply initial state
if (!neonEnabled) {
    body.classList.add('reduce-motion');
    neonToggle.textContent = 'üí§ STATIC';
}

neonToggle.addEventListener('click', function() {
    body.classList.toggle('reduce-motion');
    
    if (body.classList.contains('reduce-motion')) {
        neonToggle.textContent = 'üí§ STATIC';
        localStorage.setItem('neonEffectsEnabled', 'false');
    } else {
        neonToggle.textContent = '‚ö° NEON';
        localStorage.setItem('neonEffectsEnabled', 'true');
    }
});

Skills shown: LocalStorage, media queries, accessibility, event handling, state management
